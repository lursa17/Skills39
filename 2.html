  1. Use VLANs on the switch (e.g., VLAN 10 for PC0, VLAN 20 for PC1).

‚öôÔ∏è 1. Configure the Switch for VLANs and Trunking

bash
enable
configure terminal
hostname Switch1

! Create VLANs
vlan 10
 name VLAN10-PCs
exit
vlan 20
 name VLAN20-PCs
exit

! Assign ports to VLANs
interface fastEthernet 0/1
 switchport mode access
 switchport access vlan 10
exit

interface fastEthernet 0/2
 switchport mode access
 switchport access vlan 20
exit

! Configure trunk link to router
interface gigabitEthernet 0/1
 switchport mode trunk
exit

! Set management IP on VLAN1 (optional)
interface vlan 1
 ip address 192.168.99.2 255.255.255.0
 no shutdown
exit

ip default-gateway 192.168.99.1
exit
write memory

‚öôÔ∏è 2. Configure the Router (Router-on-a-Stick, DHCP, and EIGRP)

bash
enable
configure terminal
hostname Router1

! Subinterfaces for VLAN 10
interface gigabitEthernet 0/0.10
 encapsulation dot1Q 10
 ip address 192.168.10.1 255.255.255.0
exit

! Subinterfaces for VLAN 20
interface gigabitEthernet 0/0.20
 encapsulation dot1Q 20
 ip address 192.168.20.1 255.255.255.0
exit

! Enable main interface
interface gigabitEthernet 0/0
 no shutdown
exit

! Configure DHCP for VLAN 10
ip dhcp pool VLAN10
 network 192.168.10.0 255.255.255.0
 default-router 192.168.10.1
 dns-server 8.8.8.8
exit

! Configure DHCP for VLAN 20
ip dhcp pool VLAN20
 network 192.168.20.0 255.255.255.0
 default-router 192.168.20.1
 dns-server 8.8.8.8
exit

! Exclude gateway addresses
ip dhcp excluded-address 192.168.10.1
ip dhcp excluded-address 192.168.20.1

! Enable EIGRP
router eigrp 100
 network 192.168.10.0
 network 192.168.20.0
 no auto-summary
exit

write memory
